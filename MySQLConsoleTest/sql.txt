select *
 FROM (
 SELECT tta.ACCOUNTID AS Account_ID, tta.ACCOUNTNAME AS Account_Name, IFNULL(css.UUID,'') AS Speak_ID, IFNULL(tta.IMAGE,'') AS Head_Path, IFNULL(tsi.T_T_P_NAME,'') AS Real_Name, IFNULL(tcm.UUID,'') AS Class_ID, IFNULL(tcm.CLASSNAME,'') AS Class_Name, IFNULL(css.T_ANSWER_COUNT,0) AS ANSWER_COUNT, IFNULL(css.T_STATUS,'0') AS IsSpeak, IFNULL(cqs.Question_Count,0) AS Question_Count
 FROM T_TBKT_STUDENT_INFO tsi
 INNER JOIN T_EDGE_SYS_ACCOUNT tta
 ON tsi.UUID = tta.TA_PERSON_ID
 LEFT JOIN (SELECT *
 FROM T_TBKT_COURSE_SPEAK_STUDENT
 WHERE TIME_TABLE_ID ='165c538f5fe84d83911d12930f2c118b') css
 ON tta.ACCOUNTID = css.T_STUDENT_ID
 LEFT JOIN (
 SELECT T_STUDENT_ID,COUNT(T_STUDENT_ID) AS Question_Count
 FROM T_TBKT_COURSE_QUESTION_STUDENT
 WHERE T_STATUS = 0 AND TIME_TABLE_ID = '165c538f5fe84d83911d12930f2c118b' GROUP BY T_STUDENT_ID ) cqs
 ON tta.ACCOUNTID = cqs.T_STUDENT_ID
 LEFT JOIN T_TBKT_CLASS_MANAGER tcm
 ON tsi.T_T_P_BJ = tcm.UUID
 WHERE FIND_IN_SET(tsi.T_T_P_BJ,'a5e94b9e01e144318c5a543cae011eed,9c9cfdd9ca1e4ec0978871d3f4ab88bc') 
  LIMIT {0},{1}
 ) m ORDER BY IsSpeak DESC,Question_Count DESC,ANSWER_COUNT DESC,Account_Name ASC
